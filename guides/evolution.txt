1.3 Hot-Reloadable Registry
typescript
class DomainRegistry {
  private domains: Map<string, DomainAdapter> = new Map();
  
  async loadFromDatabase() {
    const domains = await prisma.domain.findMany({
      include: { dataSources: { where: { enabled: true } } }
    });
    
    for (const domain of domains) {
      const adapter = this.buildAdapter(domain);
      this.domains.set(domain.name, adapter);
    }
  }
  
  async reloadDomain(domainName: string) {
    const domain = await prisma.domain.findUnique({
      where: { name: domainName },
      include: { dataSources: { where: { enabled: true } } }
    });
    
    if (domain) {
      this.domains.set(domainName, this.buildAdapter(domain));
    } else {
      this.domains.delete(domainName);
    }
  }
  
  private buildAdapter(domain: DomainWithSources): DomainAdapter {
    const hasPersistent = domain.dataSources.some(ds => ds.storeResults === true);
    
    if (!hasPersistent) {
      return new EphemeralAdapter(domain);
    }
    
    return new GenericAdapter(domain); // Writes to query_results
  }
}

async function registerDomain(discoveryId: string, overrides?: Overrides) {
  const record = await prisma.domainDiscovery.findUnique({
    where: { id: discoveryId }
  });
  
  const config = { ...record.proposedConfig, ...overrides };
  
  if (config.storeResults === false) {
    // Ephemeral path — no storage
    const domain = await prisma.domain.upsert({
      where: { name: config.name },
      create: { name: config.name, intentPatterns: config.intents, ... },
      update: {}
    });
    
    for (const source of config.sources) {
      await prisma.dataSource.create({
        data: {
          domainId: domain.id,
          name: source.name,
          url: source.url,
          type: source.type,
          fieldMap: source.fieldMap,
          storeResults: false,
          refreshPolicy: "realtime"
        }
      });
    }
    
    await registry.reloadDomain(domain.name);
    
  } else {
    // Persistent path — uses query_results table
    // Same domain creation, but storeResults = true
    // Adapter will write to query_results on fetch
  }
  
  await prisma.domainDiscovery.update({
    where: { id: discoveryId },
    data: { status: "registered" }
  });
}

2.3 Output Router
typescript
class OutputRouter {
  private generators: OutputGenerator[] = [];
  
  register(generator: OutputGenerator) {
    this.generators.push(generator);
  }
  
  async route(query: ParsedIntent, data: QueryResult | null, context: GeneratorContext) {
    // Check if user explicitly requested an output type
    if (query.output_type) {
      const generator = this.generators.find(g => g.type === query.output_type);
      if (generator) return generator.generate(query, data, context);
    }
    
    // Otherwise, score all generators
    const scored = await Promise.all(
      this.generators.map(async g => ({
        generator: g,
        score: await g.canHandle(query, data, context) ? 1 : 0
      }))
    );
    
    const best = scored.sort((a, b) => b.score - a.score)[0];
    return best?.generator.generate(query, data, context);
  }
}

3.2 Travel Time Service with Caching
typescript
class TravelTimeService {
  private cache: Map<string, TravelTimeCache> = new Map();
  
  async calculate(origin: string, destination: string, mode: string) {
    const key = `${origin}:${destination}:${mode}`;
    
    if (this.cache.has(key)) {
      const cached = this.cache.get(key);
      if (cached && cached.expiresAt > new Date()) {
        return cached;
      }
    }
    
    const result = await googleMaps.distanceMatrix({
      origins: [origin],
      destinations: [destination],
      mode
    });
    
    this.cache.set(key, {
      duration: result.duration,
      distance: result.distance,
      expiresAt: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) // 30 days
    });
    
    return result;
  }
  
  // Precompute for known domains (Edinburgh Fringe venues, etc.)
  async precomputeForDomain(domainName: string) {
    const venues = await prisma.queryResults.findMany({
      where: { domain_name: domainName },
      distinct: ['location'],
      select: { location: true }
    });
    
    for (const v1 of venues) {
      for (const v2 of venues) {
        await this.calculate(v1.location, v2.location, 'walking');
      }
    }
  }
}

4.2 Agent Execution Loop
typescript
// src/agent/execution.ts

async function executeWithAgent(query: string, context: AgentContext) {
  const sessionId = context.sessionId || crypto.randomUUID();
  
  const response = await reasoningAgent.run({
    messages: [
      { role: 'system', content: reasoningAgent.instructions },
      { role: 'user', content: query }
    ],
    sessionId,
    tools: Object.values(tools),
    toolChoice: 'auto'
  });
  
  // Parse tool calls from response
  const toolCalls = extractToolCalls(response);
  
  for (const call of toolCalls) {
    const tool = tools[call.name];
    if (tool) {
      const result = await tool.execute(call.parameters);
      response.addToolResult(call.id, result);
    }
  }
  
  // Continue if more tool calls
  if (response.needsMoreTools()) {
    return reasoningAgent.continue(response);
  }
  
  // Determine output type from response
  const outputType = detectOutputType(response.content);
  
  return {
    type: outputType,
    content: response.content,
    artifacts: response.artifacts,
    sessionId
  };
}

5.1 Autonomous Discovery Agent

class DiscoveryAgent {
  async discoverForQuery(query: string, context: AgentContext) {
    // Agent identifies that no existing tool/data source can answer
    if (await this.capabilityGapDetected(query)) {
      // Use existing discovery pipeline
      const sources = await discoveryPipeline.run(query);
      
      // Propose new data source or tool
      const proposal = await this.proposeCapability(sources);
      
      if (proposal.confidence > 0.9 && proposal.risk === 'low') {
        // Auto-approve for low-risk cases
        await this.autoRegister(proposal);
      } else {
        // Flag for human review
        await this.createReviewRecord(proposal);
      }
    }
  }
  
  async proposeCapability(sources: DiscoveredSource[]) {
    // LLM analyzes samples to propose new tool or data source
    const response = await llm.complete({
      prompt: `
        Based on these sample rows, propose a new capability:
        - Should this be a new data source for an existing domain?
        - Should this be a new tool for the reasoning engine?
        - What would the tool do?
        
        Sample rows: ${JSON.stringify(sources.sampleRows)}
      `
    });
    
    return {
      type: response.type, // 'data_source' or 'tool'
      domain: response.domain,
      toolDefinition: response.toolDefinition,
      confidence: response.confidence,
      risk: response.risk
    };
  }
}

5.3 Learning and Adaptation Loop
typescript
class LearningLoop {
  async processFeedback() {
    // Analyze successful and failed interactions
    const successful = await this.getSuccessfulQueries();
    const failed = await this.getFailedQueries();
    
    // Identify patterns
    const patterns = await this.identifyPatterns(successful, failed);
    
    // Update agent memory
    await reasoningAgent.updateSystemInstructions(
      this.incorporatePatterns(patterns)
    );
    
    // Suggest new tools based on repeated patterns
    for (const pattern of patterns) {
      if (pattern.frequency > 10 && !this.hasToolFor(pattern)) {
        await this.suggestNewTool(pattern);
      }
    }
  }
  
  async identifyPatterns(successful: Query[], failed: Query[]) {
    // Use LLM to cluster queries by intent and approach
    const clusters = await llm.embedAndCluster([
      ...successful.map(q => q.intent),
      ...failed.map(q => q.intent)
    ]);
    
    return clusters.map(cluster => ({
      intent: cluster.intent,
      frequency: cluster.count,
      successRate: cluster.successCount / cluster.totalCount,
      commonTools: cluster.toolsUsed,
      suggestedTool: cluster.missingCapability
    }));
  }
}

Conclusion
This evolution path transforms DREDGE from a domain-specific data retrieval system into a general-purpose intelligent reasoning engine. Each phase is independently valuable:

Phase 1 delivers the zero-code domain addition that's currently missing

Phase 2 enables rich multi-modal outputs

Phase 3 creates reusable primitives that unlock future capabilities

Phase 4 adds true intelligence with Mastra agents

Phase 5 achieves autonomous capability discovery

The architecture is designed for incremental adoption—you can stop at any phase and have a working system. The reasoning agent in Phase 4 can use tools from Phase 3 without requiring Phase 5. The output generators from Phase 2 continue to work alongside the agent.

The key architectural insight is that intelligence emerges from composition, not special-case code. By building a rich tool library and an agent that can compose them, you solve not just the immediate use cases (Fringe itineraries, train travel) but all future reasoning tasks without additional development.
